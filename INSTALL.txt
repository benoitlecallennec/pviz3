
-. Linux/OSX
BUILD_TYPE=Release
cmake \
-DCMAKE_BUILD_TYPE:STRING=$BUILD_TYPE \
-DCMAKE_INSTALL_PREFIX:PATH=\$ENV{PWD}/install \
..

-. Windows
set PATH=%PATH%;%QT_ROOT%\bin
set BUILD_TYPE=Release
cmake -G "NMake Makefiles" -DCMAKE_BUILD_TYPE:STRING=%BUILD_TYPE% -DCMAKE_INSTALL_PREFIX:PATH=%CD%/install -DCMAKE_PREFIX_PATH:PATH=%QT_ROOT%;%VTK_ROOT%;%BOOST_ROOT% ..
